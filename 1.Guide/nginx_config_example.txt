# Đây là file cấu hình Nginx *khởi đầu* cho ditavn.tababyco.com
# Bạn nên sử dụng cấu hình này TRƯỚC KHI chạy Certbot.
# Certbot sẽ tự động nâng cấp file này lên phiên bản HTTPS.

server {
    # Lắng nghe trên cổng 80 cho các kết nối HTTP thông thường
    listen 80;
    # (Tùy chọn) Lắng nghe trên IPv6
    # listen [::]:80;

    # Đặt thư mục gốc cho ứng dụng frontend đã được build
    # LƯU Ý: Hãy thay thế đường dẫn này bằng đường dẫn chính xác đến thư mục dự án của bạn
    root /var/www/Ditavn-Ads-Auto/dist;
    index index.html;

    # Đặt tên máy chủ là tên miền của bạn
    server_name ditavn.tababyco.com;

    # Cấu hình cho các ứng dụng trang đơn (SPA) như React
    location / {
        try_files $uri $uri/ /index.html;
    }

    # Chuyển tiếp các yêu cầu API đến backend Node.js
    location /api {
        # Tăng thời gian chờ để xử lý các yêu cầu AI chạy lâu
        proxy_read_timeout 300s;
        proxy_send_timeout 300s;
        proxy_connect_timeout 75s;

        proxy_pass http://localhost:4003; # Đảm bảo cổng này khớp với cổng backend của bạn
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
    }
}
